# 代码随想录刷题 剑指Offer 29.顺时针打印矩阵 √
```c++ {.line-numbers}
class Solution {
public:
    vector<int> spiralArray(vector<vector<int>>& array) {
        if(array.size() == 0 || array[0].size() == 0)
            return {};

        vector<int> ans;
        int m = array.size(), n = array[0].size();
        int loop = 0;
        int i, j;
        int cnt = 0, total = m * n;
        while(loop <= n/2){
            for(i = loop, j = loop; j <= n-loop-1; j++){
                if(cnt >= total)
                    break;
                ans.push_back(array[i][j]);
                cnt++;
            }
            for(i = i+1, j = j-1; i <= m-loop-1; i++){
                if(cnt >= total)
                    break;
                ans.push_back(array[i][j]);
                cnt++;
            }
            for(i = i-1, j = j-1; j > loop; j--){
                if(cnt >= total)
                    break;
                ans.push_back(array[i][j]);
                cnt++;
            }
            for(; i > loop; i--){
                if(cnt >= total)
                    break;
                ans.push_back(array[i][j]);
                cnt++;
            }
            loop++;
        }
        return ans;
    }
};
```
