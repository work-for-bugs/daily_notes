# 6.7
不设置全局变量，使用传参的方式，会超时
```c++ {.line-numbers}
class Solution {
public:
    unordered_map<char, int> hash_t, hash_s;
    bool check(){
        //比较t哈希表是否在s哈希表中
        for(const auto &p:hash_t){
            if(hash_s[p.first] < p.second)//动态窗口里的少了
                return false;
        }
        return true;
    }

    string minWindow(string s, string t) {
        //统计t的字符数量
        
        for(int i = 0; i < t.length(); i++){
            hash_t[t[i]]++;
        }

        //滑动窗口
        int ans = INT_MAX, ans_left = 0;
        int left = 0, right = 0;

        while(right < s.length()){
            if(hash_t.find(s[right]) != hash_t.end())
                hash_s[s[right]]++;
            while(check() && left <= right){//当前窗口包含t字符串
                if(right - left + 1 < ans){
                    ans = right - left + 1;
                    ans_left = left;
                }
                if(hash_t.find(s[left]) != hash_t.end())
                    hash_s[s[left]]--;
                left++;//收缩窗口
            }
            right++;
        }
        return ans == INT_MAX ? string() : s.substr(ans_left, ans);
    }
};
```
