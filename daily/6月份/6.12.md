# 代码随想录刷题 54. 螺旋矩阵
```c++ {.line-numbers}
class Solution {
public:
    vector<int> spiralOrder(vector<vector<int>>& matrix) {
        vector<int> ans;
        int cnt = 0;
        int m = matrix.size(), n = matrix[0].size();
        int total = m * n;
        int loop = 0;
        int i, j;
        while(loop <= m/2){
            //遍历上边的行
            for(j = loop, i = loop; j < n-loop-1; j++){
                if(cnt >= total)
                    break;
                ans.push_back(matrix[i][j]);
                cnt++;
            }
            //遍历右边的列
            for(; i < m-loop-1; i++){
                if(cnt >= total)
                    break;
                ans.push_back(matrix[i][j]);
                cnt++;
            }
            //遍历下边的行
            for(; j > loop; j--){
                if(cnt >= total)
                    break;
                ans.push_back(matrix[i][j]);
                cnt++;
            }
            //遍历左边的列
            for(; i > loop; i--){
                if(cnt >= total)
                    break;
                ans.push_back(matrix[i][j]);
                cnt++;
            }
            loop++;
        }
        if(m % 2 == 1 && m == n)//n阶矩阵 需要再添加最中心的数 因为不会进入上面的for循环
            ans.push_back(matrix[--loop][loop]);
        return ans;
    }
};
```
