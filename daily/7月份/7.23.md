## 右旋转字符串
两次反转   
```c++ {.line-numbers}
#include <iostream>
#include <algorithm>
using namespace std;

int main(){
    string s;
    int k;
    while(cin >> k >> s){
        reverse(s.begin(), s.end());
        reverse(s.begin(), s.begin()+k);
        reverse(s.begin()+k, s.end());
        cout << s << endl;
    }
    return 0;
}
```
