## 替换数字
```c++ {.line-numbers}
#include <iostream>
#include <string.h>
using namespace std;

int main(){
    string s;
    while(cin >> s){
        int n = s.size();
        int index = n - 1;
        int digitalCount = 0;
        for(int i = 0; i < n; i++){
            if(s[i] >= '0' && s[i] <= '9')
                digitalCount++;
        }
        
        s.resize(n + digitalCount * 5);
        int newIndex = s.size() - 1;
        while(index >= 0){
            if(s[index] >= '0' && s[index] <= '9'){
                s[newIndex--] = 'r';
                s[newIndex--] = 'e';
                s[newIndex--] = 'b';
                s[newIndex--] = 'm';
                s[newIndex--] = 'u';
                s[newIndex--] = 'n';
            }
            else{
                s[newIndex--] = s[index];
            }
            index--;
        }
        
        cout << s << endl;
    }
    
    return 0;
}
```
