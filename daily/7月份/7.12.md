## 202 快乐数
```c++ {.line-numbers}
class Solution {
public:
    bool isHappy(int n) {
        unordered_set<int> hash;
        int sum = 0;
        while(1){
            sum = 0;
            while(n > 0){
                int a = n % 10;
                sum += a * a;
                n = n / 10;
            }
            if(sum == 1)
                return true;
            if(hash.find(sum) != hash.end())
                return false;
            hash.insert(sum);
            n = sum;
        }

        return true;
    }
};
```
