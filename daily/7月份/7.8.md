## 赎金信
```c++ {.line-numbers}
class Solution {
public:
    bool canConstruct(string ransomNote, string magazine) {
        int hash[26] = {0};
        for(char a : magazine){
            hash[a - 'a']++;
        }
        for(char a : ransomNote){
            hash[a - 'a']--;
            if( hash[a - 'a'] < 0)
                return false;
        }
        return true;
    }
};
```
