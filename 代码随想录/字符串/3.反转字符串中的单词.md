```c++ {.line-numbers}
class Solution {
public:
    string reverseWords(string s) {
        reverse(s.begin(), s.end());

        int slow = 0, fast = 0;
        while(fast < s.size()){
            if(s[fast] != ' '){
                if(slow != 0)
                    s[slow++] = ' ';
                int i = slow;
                while(fast < s.size() && s[fast] != ' '){
                    s[slow++] = s[fast++];
                }
                int j = slow;
                reverse(s.begin()+i, s.begin()+j);
            }
            else{
                fast++;
            }
        }

        s.resize(slow);
        return s;
    }
};
```
先全反转，再逐个单词反转。
