从两头到中间 新数组的下标由大到小更新   
```c++ {.line-numbers}
vector<int> result
k = nums.size-1;
for(i = 0, j=nums.size-1; i<=j; ){
    if(nums[i] * nums[i] > nums[j] * nums[j]){
        result[k--] = nums[i] * nums[i];
        i++;
    }
    else{
        result[k--] = nums[j] * nums[j];
        j--;
    }
    return result;
}
```
