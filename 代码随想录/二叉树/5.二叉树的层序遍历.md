queue<TreeNode*> que;
if(root != NULL)
    que.push(root);
while(!que.empty()){
    size = que.size();
    vector<int> vec;
    while(size--){//当前层的节点数量，每层该值更新
        TreeNode* node = que.front();
        que.pop();
        vec.push_back(node->val);
        if(node->left != NULL)
            que.push(node->left);
        if(node->right != NULL)
            que.push(node->right);
    }
    result.push_back(vec);
}
return result;//返回了二维数组
