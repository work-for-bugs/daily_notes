stack<int> st;
for (int i = 0; i < s.size(); i++) {
    if (s[i] == "+" || s[i] == "-" || s[i] == "*" || s[i] == "/") {
        int num1 = st.top();
        st.pop();
        int num2 = st.top();
        st.pop();
        switch (s[i]) {
            case '+':
                st.push(num1 + num2);
                break;
            case '-':
                st.push(num2 - num1);
                break;
            case '*':
                st.push(num1 * num2);
                break;
            case '/':
                st.push(num2 / num1);
                break;
        }
    }
    else{
        st.push(s[i] - '0');
    }
