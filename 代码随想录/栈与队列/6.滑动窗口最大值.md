# 整体思路
单调队列   
# 单调队列模拟过程
先push前k个元素，在push的过程中，如果新加进来的元素比前面的元素大，就把前面的元素出队列，保持最大的元素在队列首部，取最大值的时候直接取队列首元素。不比前面的元素大则正常push   
# 代码实现
```c++ {.line-numbers}
deque<int> que;
pop(int val) {
    if (!que.empty() && que.front() == val) 
        que.pop_front();
}
push(int val) {
    while (!que.empty() && val > que.back())
        que.pop_back();//把所有比这个元素大的元素从后边出队列
    que.push_back(val);
}
getMaxValue() {
    return que.front();
}
```

```c++ {.line-numbers}
class Solution {
public:
    deque<int> que;
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> ans;
        for (int i = 0; i < n; i++) {
            if (i < k) {
                push(nums[i]);
            }
            else {
                ans.push_back(getMaxValue());
                pop(nums[i-k]);
                push(nums[i]);
            }
        }
        ans.push_back(getMaxValue());
        return ans;
    }

    void push(int val) {
        while (!que.empty() && que.back() < val) {
            que.pop_back();
        }
        que.push_back(val);
    }

    void pop(int val) {
        if (val == que.front())
            que.pop_front();
    }

    int getMaxValue() {
        return que.front();
    }
};
```
